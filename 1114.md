#include <cstdio>
#include <algorithm>
using namespace std;
const int MAX_N = 1e4;

struct Family {
	int id, fid, mid, cid[5];
	int num, area;
} fs[MAX_N];
int father[MAX_N];
struct Node {
	int id = 0, population = 0, num = 0, area = 0;
	double avgNum, avgArea;
	bool flag = false;
} ans[MAX_N];
int vis[MAX_N];

int find(int x) {
	while (x != father[x]) {
		x = father[x];
	}
	return x;
}

void Union(int a, int b) {
	int A = find(a);
	int B = find(b);
	if (A > B) {
		father[A] = B;
	}
	else {
		father[B] = A;
	}
}

bool cmp(Node a, Node b) {
	return a.avgArea == b.avgArea ? a.id < b.id : a.avgArea > b.avgArea;
}

int main() {
	int N;
	scanf("%d", &N);
	for (int i = 0; i < MAX_N; i++) {
		father[i] = i;
	}
	for (int n = 0; n < N; n++) {
		int K;
		scanf("%d %d %d %d", &fs[n].id, &fs[n].fid, &fs[n].mid, &K);
		vis[fs[n].id] = 1;
		if (fs[n].fid != -1) {
			Union(fs[n].id, fs[n].fid);
			vis[fs[n].fid] = 1;
		}
		if (fs[n].mid != -1) {
			Union(fs[n].id, fs[n].mid);
			vis[fs[n].mid] = 1;
		}
		for (int k = 0; k < K; k++) {
			scanf("%d", &fs[n].cid[k]);
			Union(fs[n].id, fs[n].cid[k]);
			vis[fs[n].cid[k]] = 1;
		}
		scanf("%d %d", &fs[n].num, &fs[n].area);
	}
	for (int i = 0; i < N; i++) {
		int id = find(fs[i].id);
		ans[id].id = id;
		ans[id].num += fs[i].num;
		ans[id].area += fs[i].area;
		ans[id].flag = true;
	}
	int cnt = 0;
	for (int i = 0; i < MAX_N; i++) {
		if (vis[i]) {
			ans[find(i)].population++;
		}
		if (ans[i].flag) {
			cnt++;
		}
	}
	for (int i = 0; i < MAX_N; i++) {
		if (ans[i].flag) {
			ans[i].avgNum = 1.0 * ans[i].num / ans[i].population;
			ans[i].avgArea = 1.0 * ans[i].area / ans[i].population;
		}
	}
	sort(ans, ans + MAX_N, cmp);
	printf("%d\n", cnt);
	for (int i = 0; i < cnt; i++) {
		printf("%04d %d %.3f %.3f\n", ans[i].id, ans[i].population, ans[i].avgNum, ans[i].avgArea);
	}
	return 0;
}
