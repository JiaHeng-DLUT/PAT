```c++
#include <iostream>
#include <map>
#include <numeric>
#include <vector>
using namespace std;

map<int, vector<int>> g;	// vertex -> edges

int main() {
	int N, M;
	cin >> N >> M;
	for (int m = 0; m < M; m++) {
		int a, b;
		cin >> a >> b;
		g[a].push_back(m);	// Vertex <a> belongs to edge <m>.
		g[b].push_back(m);	// Vertex <b> belongs to edge <m>.
	}
	int K;
	cin >> K;
	for (int k = 0; k < K; k++) {
		int N;
		cin >> N;
		vector<int> vis(M, 0);
		for (int n = 0; n < N; n++) {
			int x;
			cin >> x;
			for (auto it : g[x]) {
				vis[it] = 1;
			}
		}
		if (accumulate(vis.begin(), vis.end(), 0) != M) {
			cout << "No" << endl;
		}
		else {
			cout << "Yes" << endl;
		}
	}
	return 0;
}
```
