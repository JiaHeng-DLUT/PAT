```c++
#include <iostream>
#include <map>
#include <set>
using namespace std;

int k;
const int max_k = 100 + 5;
string s;
map<char, bool> m;
set<char> res;

int main() {
	cin >> k >> s;
	int num = 1;
	for (int i = 1; i <= s.size(); i++) {
		if (i == s.size() || s[i] != s[i - 1]) {
			if (num % k != 0) {
				m[s[i - 1]] = false;
			}
			else {
				m[s[i - 1]] = true;
			}
			num = 1;
		}
		else {
			num++;
		}
	}
	for (int i = 0; i < s.size(); i++) {
		if (m[s[i]] && res.count(s[i]) == 0) {
			cout << s[i];
			res.insert(s[i]);
		}
	}
	cout << endl;
	for (int i = 0; i < s.size(); i++) {
		cout << s[i];
		if (m[s[i]]) {
			i += k - 1;
		}
	}
	cout << endl;
	return 0;
}
```
