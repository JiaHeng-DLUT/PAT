## 1 Key Points

- **DFS with Recursion**

## 2 Answers

```c++
#include <iostream>
#include <vector>
using namespace std;
const int MAX_N = 500 + 5;
const int INF = 1000000000;

// input
int cmax, N, sp, M;
int d[MAX_N][MAX_N];
int bikes[MAX_N];
// output
// c: current
vector<int> shortestPath, cPath;
int minDistance = INF, minSend = INF, minTakeBack = INF;
int cDistance = 0, cSend = 0, cTakeBack = 0;
bool visited[MAX_N] = { 0 };

void dfs(int c) {
	if (cDistance > minDistance) {
		return;
	}
	if (c == sp) {
		if (cDistance < minDistance || (cDistance == minDistance && cSend < minSend) || (cDistance == minDistance && cSend == minSend && cTakeBack < minTakeBack)) {
			shortestPath = cPath;
			minDistance = cDistance;
			minSend = cSend;
			minTakeBack = cTakeBack;
			return;
		}
	}
	for (int i = 1; i < N; i++) {
		if (!visited[i] && d[c][i] < INF) {
			cPath.push_back(i);
			cDistance += d[c][i];
			int lastSend = cSend;
			int lastTakeBack = cTakeBack;
			if (bikes[i] + cTakeBack < cmax / 2) {
				cSend += cmax / 2 - bikes[i] - cTakeBack;
				cTakeBack = 0;
			}
			else {
				cTakeBack = -cmax / 2 + bikes[i] + cTakeBack;
			}
			visited[i] = 1;
			dfs(i);
			visited[i] = 0;
			cTakeBack = lastTakeBack;
			cSend = lastSend;
			cDistance -= d[c][i];
			cPath.pop_back();
		}
	}
}

int main() {
	// input
	cin >> cmax >> N >> sp >> M;
	// cout << cmax << " " << N << " " << sp << " " << M << endl;
	N++;
	bikes[0] = 0;
	for (int i = 1; i < N; i++) {
		cin >> bikes[i];
	}
	/*
	for (int i = 0; i < N; i++) {
		cout << t[i] << " ";
	}
	cout << endl;
	// */
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < N; j++) {
			d[i][j] = INF;
		}
	}
	for (int i = 0; i < M; i++) {
		int a, b, c;
		cin >> a >> b >> c;
		d[a][b] = d[b][a] = c;
	}
	/*
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < N; j++) {
			cout << d[i][j] << " ";
		}
		cout << endl;
	}
	// */
	// solve
	dfs(0);
	// output
	cout << minSend << " 0";
	for (int i = 0; i < shortestPath.size(); i++) {
		cout << "->" << shortestPath[i];
	}
	cout << " " << minTakeBack << endl;
	return 0;
}
```
