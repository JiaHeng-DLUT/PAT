# 1053

- [1053 Path of Equal Weight (30 point(s))](https://pintia.cn/problem-sets/994805342720868352/problems/994805424153280512)
- `dfs` + `sort`

## Solution 1: dfs -> sort

![Accepted](https://i.loli.net/2019/08/05/ZVRUElLnvWHNomF.png)

```c++
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
const int MAX_N = 100 + 5;

// input
int N; // 0 < N <= 100
int M; // 0 <= M < N
int S; // 0 < S < 2^30
int W[MAX_N]; // 0 < W[i] < 1000
// solve
int tree[MAX_N][MAX_N] = { 0 }; // tree[i][j] = 0 / 1
int nonLeaf[MAX_N] = { 0 }; // nonLeaf[i] = 0 / 1
// output
vector<vector<int>> res;
vector<int> v1;

void dfs(int c, int s) {
	for (int i = 0; i < N; i++) {
		if (tree[c][i]) {
			v1.push_back(W[i]);
			s -= W[c];
			dfs(i, s);
			s += W[c];
			v1.pop_back();
		}
	}
	if (!nonLeaf[c] && s == W[c]) {
		res.push_back(v1);
	}
}

bool comp(const vector<int>& v1, const vector<int>& v2) {
	for (int i = 0; i < min(v1.size(), v2.size()); i++) {
		if (v1[i] < v2[i]) {
			return false;
		}
		else if (v1[i] > v2[i]) {
			return true;
		}
	}
	return v1.size() > v2.size();
}

int main() {
	// input
	cin >> N >> M >> S;
	for (int i = 0; i < N; i++) {
		cin >> W[i];
	}
	for (int i = 0; i < M; i++) {
		int ID, K;
		cin >> ID >> K;
		nonLeaf[ID] = 1;
		for (int j = 0; j < K; j++) {
			int ID1;
			cin >> ID1;
			tree[ID][ID1] = 1;
		}
	}
	/*
	// test input
	for (int i = 0; i < N; i++) {
		cout << W[i] << " ";
	}
	cout << endl;
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < N; j++) {
			cout << tree[i][j] << " ";
		}
		cout << endl;
	}
	// */
	// solve
	dfs(0, S);
	sort(res.begin(), res.end(), comp);
	// output
	for (auto vec : res) {
		cout << W[0];
		for (auto val : vec) {
			cout << " " << val;
		}
		cout << endl;
	}
}
```

## Solution 2: sort -> dfs

![Accepted](https://i.loli.net/2019/08/05/AF7jYTJogHN4nbt.png)

```c++
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
const int MAX_N = 100 + 5;

struct node {
	int no, weight;
	vector<int> child;
} Node[MAX_N];
int s;
vector<int> temp;

bool cmp(int a, int b) {
	return Node[a].weight > Node[b].weight;
}

void dfs(int x, int sum) {
	if (sum > s || (sum == s && Node[x].child.size() > 0)) {
		return;
	}
	if (Node[x].child.size() == 0 && sum == s) {
		temp.push_back(x);
		for (int i = 0; i < temp.size(); ++i) {
			printf("%d", Node[temp[i]].weight);
			if (i < temp.size() - 1) {
				printf(" ");
			}
			else {
				printf("\n");
			}
		}
		temp.pop_back();
		return;
	}
	temp.push_back(x);
	for (int i = 0; i < Node[x].child.size(); ++i) {
		dfs(Node[x].child[i], sum + Node[Node[x].child[i]].weight);
	}
	temp.pop_back();
}

int main() {
	int n, m, k, no, t;
	scanf("%d %d %d", &n, &m, &s);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &Node[i].weight);
	}
	for (int i = 0; i < m; ++i) {
		scanf("%d %d", &no, &k);
		for (int j = 0; j < k; ++j) {
			scanf("%d", &t);
			Node[no].child.push_back(t);
		}
		sort(Node[no].child.begin(), Node[no].child.end(), cmp);
	}
	dfs(0, Node[0].weight);
	return 0;
}
```
