### Case 2:

```
i: 0 1 2 3 4
N: 5 4 3 1 7
         ^
left = 543
right = 7
```

```
ans += left * pow(10, n - i - 1)
5 4 2 1 x
5 4 1 1 x
...
0 0 0 1 x
```

```
1 == 1 (5 4 3 1 x)
ans += right + 1
5 4 3 1 0
5 4 3 1 1
...
5 4 3 1 7
```

![Accepted](https://i.loli.net/2019/08/17/F6Ce92TSbgQ4oPA.png)

```c++
#include <iostream>
#include <cmath>
#include <string>
using namespace std;

string s = "";
int ans = 0;

int main() {
	cin >> s;
	int n = s.size(), N = 0;
	for (int i = 0; i < n; i++) {
		N = N * 10 + s[i] - '0';
	}
	// cout << N << endl;
	int left = 0, right = N;
	for (int i = 0; i < n; i++) {
		if (i >= 1) {
			left = left * 10 + s[i - 1] - '0';
		}
		right = right % (int)pow(10, n - i - 1);
		// cout << left << " " << right << endl;

		ans += left * pow(10, n - i - 1);

		if (s[i] == '1') {
			ans += right + 1;
		}
		else if (s[i] > '1') {
			ans += pow(10, n - i - 1);
		}
	}
	cout << ans << endl;
	return 0;
}

/*
Sample Input:
12
Sample Output:
5
*/

```
